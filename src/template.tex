% Use 'article' class, suitable for simple documents like resumes
% 12pt font size and A4 paper.
\documentclass[12pt,a4paper]{article}

% Encoding & Fonts
% UTF-8 input encoding, necessary for non-ASCII characters.
\usepackage[utf8]{inputenc}

% Output font encoding: T1 ensures proper hyphenation and accented characters.
\usepackage[T1]{fontenc}

% Sets default font to sans-serif (Helvetica)
% Good for ATS readability.
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}

% Layout
% Sets 1cm margins all around
% ATS doesnâ€™t care about margin size
% 2cm is usually safer.
\usepackage[margin=1cm]{geometry}

% Adds space between paragraphs, improves readability
\setlength{\parskip}{6pt}

% No paragraph indentation, common in resumes
\setlength{\parindent}{0pt}

% Links
% Makes links clickable and blue
% colorlinks=true is necessary for blue links instead of boxes
% citecolor is harmless here, although resumes rarely use citations
\usepackage[colorlinks=true, linkcolor=blue, urlcolor=blue, citecolor=blue]{hyperref}

% Force all links (including mailto: and tel:) to use sans-serif
\usepackage{url}
\urlstyle{same}

% Remove section numbering and page numbers
\setcounter{secnumdepth}{0}

% Remove page numbers
\pagenumbering{gobble}

% Markdown list support
% Provides control over list spacing and bullets
\usepackage{enumitem}

% Define \tightlist to avoid Pandoc error
\providecommand{\tightlist}{%
    \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Controls spacing and formatting of section/subsection headings
\usepackage{titlesec}

% Heading spacing
% Section headings: bold, bigger font
\titleformat{\section}{\Huge}{}{0em}{}

% Subsection headings: bold, slightly smaller
\titleformat{\subsection}{\LARGE}{}{0em}{}

% Subsubsection headings: normal bold, smaller
\titleformat{\subsubsection}{\large}{}{0em}{}

% Section spacing: {left}{before}{after}
\titlespacing*{\section}{0pt}{16pt}{0pt}

% Subsection spacing: {left}{before}{after}
\titlespacing*{\subsection}{0pt}{12pt}{0pt}

% Subsection spacing: {left}{before}{after}
\titlespacing*{\subsubsection}{0pt}{8pt}{0pt}

% Optional: make all text left-aligned (avoids justification stretching)
\raggedright

% Document body
% Pandoc inserts Markdown content here
\begin{document}
    $body$
\end{document}
